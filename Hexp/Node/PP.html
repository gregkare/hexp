def pp_attributes
  attrs = @node.attributes
  return '' if attrs.empty?
  ', ' + attrs.inspect
end

def pp_children
  children = @node.children
  return ']' if children.empty?
  ", [\n#{ children.map(&:pp).join(",\n") }]]"
end
